## 拖拽事件
HTML5提供了原生的拖拽事件，主要包括以下几个事件：
1. `dragstart`：当用户开始拖动一个元素时触发。
2. `drag`：当元素被拖动时持续触发。
3. `dragend`：当用户释放鼠标按钮，结束拖动时触发。
4. `dragenter`：当被拖动的元素进入一个可放置的目标元素时触发。
5. `dragover`：当被拖动的元素在一个可放置的目标元素上持续移动时触发。
6. `dragleave`：当被拖动的元素离开一个可放置的目标元素时触发。
7. `drop`：当被拖动的元素在一个可放置的目标元素上释放鼠标按钮时触发。

## 代码
```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>拖拽事件 Demo</title>
    <style>
        .container {
            display: flex;
            justify-content: space-between;
            padding: 20px;
        }

        .left,
        .right {
            width: 300px;
            height: 300px;
            border: 2px dashed #999;
            border-radius: 8px;
            padding: 10px;
            transition: background-color 0.2s;
        }

        .left {
            border-color: blue;
        }

        .right {
            border-color: greenyellow;
        }

        .right.drag-over {
            background-color: rgba(173, 255, 47, 0.2);
        }

        .box {
            width: 100px;
            height: 100px;
            background-color: red;
            color: white;
            display: flex;
            align-items: center;
            justify-content: center;
            border-radius: 6px;
            cursor: grab;
            user-select: none;
        }

        .box:active {
            cursor: grabbing;
            opacity: 0.7;
        }

        .right .box {
            background-color: darkred;
        }

        h2 {
            text-align: center;
            font-family: sans-serif;
        }

        .info {
            text-align: center;
            font-family: sans-serif;
            color: #666;
            margin-top: 20px;
        }

        #log {
            margin-top: 20px;
            padding: 10px;
            background: #f5f5f5;
            border-radius: 6px;
            font-size: 12px;
            font-family: monospace;
            max-height: 200px;
            overflow-y: auto;
        }
    </style>
</head>
<body>

    <h2>拖拽事件 Demo</h2>
    <p class="info">将红色方块拖拽到右侧绿色边框区域中</p>

    <div class="container">
        <div class="left">
            <div class="box" draggable="true">被拖拽元素</div>
        </div>

        <div class="right">
            目标元素
        </div>
    </div>

    <div class="info">
        <strong>事件日志：</strong>
        <button onclick="document.getElementById('log').innerHTML=''">清空日志</button>
    </div>
    <div id="log"></div>

    <script>
        const box = document.querySelector('.box')
        const right = document.querySelector('.right')
        const logEl = document.getElementById('log')

        function log(msg) {
            const time = new Date().toLocaleTimeString()
            logEl.innerHTML += `[${time}] ${msg}<br>`
            logEl.scrollTop = logEl.scrollHeight
        }

        /*
         * 1. 开启拖拽（HTML 中设置 draggable="true"）
         * 2. 获取被拖拽元素
         */
        var target = null

        /* ===== 拖拽元素相关事件 ===== */

        /* 开始拖拽事件 */
        box.addEventListener('dragstart', (e) => {
            log(`dragstart → target: ${e.target.className}`)
            target = e.target
            e.dataTransfer.effectAllowed = 'move'
        })

        /* 持续拖拽事件 */
        box.addEventListener('dragover', (e) => {
            e.preventDefault()
            // log('dragover on box')
        })

        /* 鼠标离开被拖拽元素事件 */
        box.addEventListener('dragleave', (e) => {
            // log('dragleave on box')
        })

        /* 鼠标松开被拖拽元素的事件 */
        box.addEventListener('dragend', (e) => {
            log(`dragend`)
        })

        /* ===== 目标元素相关事件 ===== */

        /* 进入目标元素事件 */
        right.addEventListener('dragenter', (e) => {
            e.preventDefault()
            log(`dragenter → 进入右侧目标区域`)
            right.classList.add('drag-over')
        })

        /* 在目标元素上持续拖拽 —— 必须阻止默认行为，否则 drop 不会触发 */
        right.addEventListener('dragover', (e) => {
            e.preventDefault()
            e.dataTransfer.dropEffect = 'move'
            // log('right dragover')
        })

        /* 离开目标元素事件 */
        right.addEventListener('dragleave', (e) => {
            // 仅在真正离开 right 区域时移除高亮
            if (!right.contains(e.relatedTarget)) {
                log(`dragleave → 离开右侧目标区域`)
                right.classList.remove('drag-over')
            }
        })

        /*
         * drop 事件触发的前提条件：
         * 必须在 dragover 中调用 e.preventDefault()
         */
        right.addEventListener('drop', (e) => {
            e.preventDefault()
            log(`drop → 在目标区域释放，将元素移入`)
            right.classList.remove('drag-over')
            if (target) {
                // 移除"目标元素"提示文字
                const textNode = right.childNodes[0]
                if (textNode && textNode.nodeType === Node.TEXT_NODE) {
                    textNode.textContent = ''
                }
                right.appendChild(target)
                log(`✅ 元素已成功移入右侧区域`)
            }
        })

        // 左侧区域也支持将元素拖回
        const left = document.querySelector('.left')
        left.addEventListener('dragover', (e) => {
            e.preventDefault()
            e.dataTransfer.dropEffect = 'move'
        })
        left.addEventListener('drop', (e) => {
            e.preventDefault()
            log(`drop → 元素拖回左侧区域`)
            if (target) {
                left.appendChild(target)
            }
        })
    </script>

</body>
</html>
```
