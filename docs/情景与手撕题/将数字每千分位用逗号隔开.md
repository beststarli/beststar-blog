## 数字有小数
```js
const format = (n) => {
  let num = n.toString()
  let decimal = ''
  // 判断是否有小数
  decimal = num.indexOf('.') !== -1 ? num.split('.')[1] : decimal
  let length = num.length
  if (length <= 3) {
    return num
  } else {
    let temp = ''
    let remainder = length % 3
    temp = decimal ? '.' + decimal : temp
    // 不是3的倍数
    if (remainder > 0) {
      return num.slice(0, remainder) + ',' + num.slice(remainder, length).match(/\d{3}/g).join(',') + temp
    // 是3的整数倍
    } else { 
      return num.slice(0, length).match(/\d{3}/g).join(',') + temp 
    }
  }
}

format(123456789.123) // 123,456,789.123
format(123456789) // 123,456,789
format(1234567.123) // 1,234,567.123
```

## 数字无小数
```js
let format = n => {
    let num = n.toString() 
    let length = num.length
    if (length <= 3) {
        return num
    } else {
        let remainder = length % 3
        if (remainder > 0) { // 不是3的整数倍
            return num.slice(0, remainder) + ',' + num.slice(remainder, length).match(/\d{3}/g).join(',') 
        } else { // 是3的整数倍
            return num.slice(0, length).match(/\d{3}/g).join(',') 
        }
    }
}
format(1232323)  // '1,232,323'
```
