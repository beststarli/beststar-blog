## 代码实现
```js
function getType(value) {
  // 判断数据是null的情况
  if (value === null) {
    return value + ''
  }

  // 判断数据是引用类型的情况
  if (typeof value === 'object') {
    let valueClass = Object.prototype.toString.call(value)
    let type = valueClass.split(' ')[1].split('')
    type.pop()
    return type.join('').toLowerCase()
  } else {
    // 判断数据是基本类型的情况
    return typeof value
  }
}
```
